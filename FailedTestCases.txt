  const req = httpMocks.createRequest();

  expect(convertedError.message).toBe(httpStatus[httpStatus.INTERNAL_SERVER_ERROR]);
  expect(convertedError.stack).toBeUndefined();
});

  const longMessage = 'a'.repeat(10000);
  const longStack = 'b'.repeat(10000);
  const error = new ApiError(httpStatus.BAD_REQUEST, longMessage, false);
  error.stack = longStack;
  const req = httpMocks.createRequest();
  const res = httpMocks.createResponse();
  const sendSpy = jest.spyOn(res, 'send');

  errorHandler(error, req, res);

  expect(sendSpy).toHaveBeenCalledWith(
    expect.objectContaining({
      code: error.statusCode,
      message: error.message,
      stack: error.stack,
    })
  );
  expect(res.locals.errorMessage).toBe(error.message);
});
